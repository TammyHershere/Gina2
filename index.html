import random # 題庫：英文 -> 中文 words = { "volume": "體積；容量", "project": "計畫；專案", "cereal": "穀物；麥片", "insurance": "保險", "course": "課程", "account": "帳戶；報告", "monument": "紀念碑", "jail": "監獄", "fable": "寓言", "glacier": "冰河", "description": "描述", "operation": "操作；運作", "executive": "主管", "category": "類別", "excerpt": "摘錄" } def play_game(): print("🎮 英文選中文小遊戲（名詞篇） 🎮") score = 0 questions = random.sample(list(words.items()), len(words)) # 打亂題目順序 for eng, correct_ch in questions: # 建立選項 options = [correct_ch] wrong = random.sample(list(words.values()), 3) for w in wrong: if w not in options: options.append(w) random.shuffle(options) # 題目 print(f"\n英文單字：{eng}") for i, opt in enumerate(options, 1): print(f"{i}. {opt}") # 玩家回答 try: choice = int(input("請輸入選項號碼：")) if options[choice - 1] == correct_ch: print("✅ 答對了！") score += 1 else: print(f"❌ 錯誤，正確答案是：{correct_ch}") except: print(f"⚠️ 輸入錯誤，正確答案是：{correct_ch}") print(f"\n🎉 遊戲結束！你的得分是 {score}/{len(words)}") if __name__ == "__main__": play_game()
